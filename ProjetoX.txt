service cloud.firestore {
  match /databases/{database}/documents {

    function isLoggedIn() {
      return request.auth != null;
    }

    function isSameTenant(resource) {
      return isLoggedIn() &&
             request.auth.token.tenantId == resource.data.tenantId;
    }

    match /users/{userId} {
      allow read, write: if request.auth.uid == userId && isSameTenant(resource);
    }

    match /plans/{planId} {
      allow read, write: if isSameTenant(resource);
    }

    match /diagnostics/{id} {
      allow read, write: if isSameTenant(resource);
    }

    match /tasks/{id} {
      allow read, write: if isSameTenant(resource);
    }

    match /simulations/{id} {
      allow read, write: if isSameTenant(resource);
    }

    match /progress/{id} {
      allow read, write: if isSameTenant(resource);
    }

    match /logs/{id} {
      allow read: if isSameTenant(resource);
      allow write: if isLoggedIn();
    }
